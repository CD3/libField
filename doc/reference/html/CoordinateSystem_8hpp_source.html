344  template<int II, typename IND, typename C, typename... Args>
345  void nearest_imp(IND& ind, C c, Args... args) const
346  {
347  // special cases:
348  // coordinate is less than smallest
349  if (c < axes[II]->operator[](0)) {
350  ind[II] = 0;
351  }
352  // coordinate is greater than largest
353  else if (c > axes[II]->operator[](axes[II]->size() - 1)) {
354  ind[II] = axes[II]->size() - 1;
355  } else {
356  // get index of element that is just below coordinate
357  ind[II] = std::upper_bound(axes[II]->begin(), axes[II]->end(), c) -
358  axes[II]->begin() - 1;
359  // now determine if coordinate is closer to the upper bound or lower bound
360  // coord - lower bound divided by upper bound minus lower bound gives the
361  // dimensionless coordinate between 0 and 1. if this is less than 0.5, we
362  // want to return the lower bound index. if itis greater than 0.5, we want
363  // to return the upper bound index don't have to worry about special cases
364  // above here.
365  ind[II] +=
366  2 * (c - axes[II]->operator[](ind[II])) /
367  (axes[II]->operator[](ind[II] + 1) - axes[II]->operator[](ind[II]));
368  }
369 
370  nearest_imp<II + 1>(ind, args...);
371  }
