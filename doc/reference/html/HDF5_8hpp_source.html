151  * and is read directly into the field data. Coordinates are set to integer indices.
152  *
153  * This function is useful for reading data written by some other application into a field.

156 void hdf5read(H5::DataSet& dset, Field<FT, N, CT>& f)
157 {

162  ") do not match the field being read into (" +
163  std::to_string(N) + ").");
164  hsize_t ddims[N];
165  dspace.getSimpleExtentDims(ddims);

169

171 
172  dset.read(f.data(), detail::get_hdf5_dtype_for_type<FT>());
